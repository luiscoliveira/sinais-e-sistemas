# define signal x(n)
def x(n: np.array) -> np.array:
	return np.piecewise(n.astype(float),
		[n<-4, ((n>=-4) & (n<0)), ((n>=0) & (n<3)), (n==3), (n>=4)],
		[0, lambda n: 0.5*n+1.5, 1, 0.5, 0])
